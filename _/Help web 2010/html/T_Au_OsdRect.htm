OsdRect Class

Examples
Copy
using(var x = new OsdRect()) {
    x.Rect = (300, 300, 100, 100);
    x.Color = Color.SlateBlue;
    x.Thickness = 4;
    x.Show();
    for(int i = 0; i < 5; i++) {
        250.ms();
        x.Visible = !x.Visible;
    }
}
