Remarks
Can be used in possibly very frequently called functions to avoid much string garbage and frequent garbage collections. For example, Wnd.ClassName and Wnd.Name use this. They are called by Wnd.Find for each window (can be hundreds of them). And WaitFor.WindowExists calls Wnd.Find every 30 or so milliseconds. It could create megabytes of garbage (window classname and name strings) in a few seconds. Every Wnd.Find call creates almost the same set of strings, therefore it makes sense to cache them. To allow GC free the cached strings, use a WeakReferenc<StringCache> object. See example.
Example
Copy 
 internal static unsafe string GetCachedString(char* p, int len)
 {
 	lock(_cacheWR) {
 		if(!_cacheWR.TryGetTarget(out var cache)) _cacheWR.SetTarget(cache = new StringCache());
 		return cache.Add(p, len);
 	}
 }
 static WeakReference<StringCache> _cacheWR = new WeakReference<StringCache>(null);
 
 //When char[] is used as text buffer.
 internal static unsafe string GetCachedString(char[] a, int len)
 {
 	Debug.Assert(len < a.Length);
 	fixed (char* p = a) return GetCachedString(p, len);
 }
